#!/usr/bin/env python3
import json
import time
import logging
from kafka import KafkaProducer
from generator import HomeworkGenerator

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

KAFKA_BOOTSTRAP_SERVERS = ['localhost:9092']
TOPIC_NAME = 'homework-control'

class HomeworkProducer:
    def __init__(self, bootstrap_servers, topic):
        self.topic = topic
        self.producer = None
        self.bootstrap_servers = bootstrap_servers

    def connect(self):
        try:
            self.producer = KafkaProducer(
                bootstrap_servers=self.bootstrap_servers,
                value_serializer=lambda v: json.dumps(v, ensure_ascii=False).encode('utf-8'),
                acks='all',
                retries=3
            )
            logger.info(f"Connected to Kafka at {self.bootstrap_servers}")
            return True
        except Exception as e:
            logger.error(f"Failed to connect: {e}")
            return False

    def send_message(self, message):
        if not self.producer:
            return False
        try:
            future = self.producer.send(self.topic, value=message)
            record_metadata = future.get(timeout=10)
            print(f"\n📤 SENT: {json.dumps(message, indent=2, ensure_ascii=False)}")
            return True
        except Exception as e:
            logger.error(f"Error: {e}")
            return False

    def close(self):
        if self.producer:
            self.producer.flush()
            self.producer.close()

def main():
    producer = HomeworkProducer(KAFKA_BOOTSTRAP_SERVERS, TOPIC_NAME)
    if not producer.connect():
        return

    logger.info(f"Starting producer. Topic: {TOPIC_NAME}")
    logger.info("Press Ctrl+C to stop...")

    try:
        while True:
            homework_data = HomeworkGenerator.generate_homework()
            producer.send_message(homework_data)
            time.sleep(2)
    except KeyboardInterrupt:
        logger.info("Producer stopped")
    finally:
        producer.close()

if __name__ == "__main__":
    main()
